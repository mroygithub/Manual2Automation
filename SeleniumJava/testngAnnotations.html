Different Types of Annotations present in TestNG

@BeforeSuite

@BeforeSuite executes before any other test/annotation/module. This annotation plays an importatnt role to set up environments/select input datatable / select execution browser / create report path etc. 


        package TestNGPKG;
        
        import org.testng.annotations.BeforeSuite;
        
        public class testNGClass {
        
        
            @BeforeSuite
            
            public void  setup()
            {
                // Select Chrome as execution Browsers....
            }
        
        
        }

@BeforeClass

@BeforeClass : The annotated method will be run before the first test method in the current class is invoked. 


    package TestNGPKG;
    
    import org.testng.annotations.*;
    
    public class testNGClass {
                                                      
        @BeforeClass
    
        public void before_class()
        {
            System.out.println("This method is executed first");
        }
        @Test
        public void testexecutoin1()
        {
            System.out.println("Test Case1");
        }
        @Test
        public void testexecutoin2()
        {
            System.out.println("Test Case2");
        }
        @Test
        public void testexecutoin3() {
            System.out.println("Test Case3");
        }
    
        }

@BeforeGroups

@BeforeGroups: This annotation method will be run only once before all the test methods belonging to a specified group. 

    package TestNGPKG;
    
    import org.testng.annotations.*;
    
    public class testNGClass {
        
    
            @BeforeGroups("Examples")
        
            public void before_class()
            {
                System.out.println("This method is executed for Groups");
            }
            @Test
            public void testexecutoin1()
            {
                System.out.println("Test Case1");
            }
            @Test(groups = "Examples")
            public void testexecutoin2()
            {
                System.out.println("Test Case2");
            }
            @Test(groups = "Examples")
            public void testexecutoin3() {
                System.out.println("Test Case3");
            }
    
        }

@AfterSuite

@AfterSuite executes after all other test/annotation/module. This annotation plays an importatnt role to create final test report/closing application etc. 

        package TestNGPKG;
        
        import org.testng.annotations.*;
        
        public class testNGClass {
        
        
                public void before_class()
                {
                    System.out.println("This method is executed for Groups");
                }
                @Test
                public void testexecutoin1()
                {
                    System.out.println("Test Case1");
                }
                @Test(groups = "Examples")
                public void testexecutoin2()
                {
                    System.out.println("Test Case2");
                }
                @Test(groups = "Examples")
                public void testexecutoin3() {
                    System.out.println("Test Case3");
                }
        
                @AfterSuite
                public void executionDone(){
                    System.out.println("Execution is Completed...");
