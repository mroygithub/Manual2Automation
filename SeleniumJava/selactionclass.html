Selenium WebDriver Action Class

We need Action class for Selenium WebDriver for below reasons:


Reusability
Code Readability
Less Code Maintenance
Exception Handling

isElementDisplayed : Action Class

        
        // Check For Element Displayed or not
        
        
        public String checkForWebElement(WebDriver driver , By by , String elementName)
        {
            String elementdisplayed = elementName+" "+"is not displayed";
        
            try{
        
                if(driver.findElement(by).isDisplayed())
                {
                    elementdisplayed = elementName+" "+"is displayed";
                }
        
            }
            catch (Exception e){}
        
            return elementdisplayed;
        }

Key Point to Note :
1.Try/Catch is used for exception handle
2.checkForWebElement method is written in Action class file.
3.This method accepts three parameters - Webdriver , web attribute , element Name
4.String return type with success/errors.

How To use it from Test Scripts?

        
        package SelJava;
        
        import org.openqa.selenium.By;
        import org.openqa.selenium.WebDriver;
        import org.openqa.selenium.chrome.ChromeDriver;
        import org.testng.Assert;
        
        public class TestScripts extends ActionClass {
        
            public  WebDriver driver = null;
        
            public static void main(String args[]) throws Exception
        
            {
                TestScripts obj = new TestScripts();
                obj.googleApplicationValidation();
        
            }
        
            public void googleApplicationValidation()
            {
                System. setProperty("webdriver.chrome.driver", "/Users/path/Downloads/chromedriver2");
                driver = new ChromeDriver();
                driver.get("https://www.google.com");
                System.out.println(checkForWebElement(driver,By.id("hplogo"),"Google Logo"));
                driver.quit();
            }
        
        }

    // Type in a Text Box
    public String typeTextBox(WebDriver driver , By by , String objectName , String textToType)
    {
        String elementdisplayed = objectName+" "+" Text Box is not displayed";
        try{
            if(driver.findElement(by).isDisplayed())
            {
                driver.findElement(by).sendKeys(textToType);
                elementdisplayed = objectName+" "+"Text Box is displayed";
            }
        }
        catch (Exception e){}
        return elementdisplayed;
    }

Key Point to Note :
1.Try/Catch is used for exception handle
2.typeTextBox method is written in Action class file.
3.This method accepts four parameters - Webdriver , web attribute , element Name , textToType
4.String return type with success/errors.

Element.Click(): Action Class



    // Click using selenium and wait [Explicit]
    public String clickWithExplicitWait(WebDriver driver , By by , String objectName , int timeTowait)
    {
        String elementdisplayed = objectName+" "+"is not clickable";
        try{
            if(driver.findElement(by).isDisplayed())
            {
                driver.findElement(by).click();
                Thread.sleep(timeTowait);
                elementdisplayed = objectName+" "+"is clickable ";
            }
        }
        catch (Exception e){}
        return elementdisplayed;
    }
    
    
    // Click using selenium and wait [Implicit]
    public String clickWithImplilicitWait(WebDriver driver , By by , String objectName , int timeTowait)
    {
        String elementdisplayed = objectName+" "+"is not clickable";
        try{
            if(driver.findElement(by).isDisplayed())
            {
                driver.findElement(by).click();
                driver.manage().timeouts().implicitlyWait(timeTowait,TimeUnit.SECONDS) ;
                elementdisplayed = objectName+" "+"is clickable ";
            }
        }
        catch (Exception e){}
        return elementdisplayed;
    }

    package SelJava;
    
    import org.openqa.selenium.WebDriver;
    import org.openqa.selenium.By;
    import org.openqa.selenium.WebElement;
    import org.openqa.selenium.support.ui.Select;
    
    import java.util.concurrent.TimeUnit;
    
    public class ActionClass {
    
        private WebDriver driver;
    
        // Check For Element Displayed or not
        public String checkForWebElement(WebDriver driver , By by , String objectName)
        {
            String elementdisplayed = objectName+" "+"is not displayed";
            try{
                if(driver.findElement(by).isDisplayed())
                {
                    elementdisplayed = objectName+" "+"is displayed";
                }
            }
            catch (Exception e){}
            return elementdisplayed;
        }
    
    
        // Type a Text Box
