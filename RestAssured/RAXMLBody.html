Rest Assured XML Body

Assume that we are getting below response by rest assured:-

{
"status": "success",
"data": {
"name": "MITHUN",
"salary": "5000",
"age": "35",
"id": 83
}
}
How to validate above json using code??
given() .header("Content-type", "application/json;charset=utf-8") .body(requestBody) .post() .then() .assertThat() .body("status", equalTo("success")) .body("data.name", equalTo("MITHUN")) .body("data.salary", equalTo("5000")) .body("data.age", equalTo("35"));
Note : We need to import below :
import static io.restassured.RestAssured.given; 

          
          package RestAssured;
          
          import static io.restassured.RestAssured.*;
          import io.restassured.matcher.RestAssuredMatchers.*;
          import static org.hamcrest.Matchers.*;
          import io.restassured.RestAssured;
          import io.restassured.response.Response;
          
          public class RestTest {
          
          
              public static void main(String[] args) {
          
          
                  RestAssured.baseURI = "http://dummy.restapiexample.com/api/v1/create";
          
                  String requestBody = "{\n" +
                          "  \"name\":\"MITHUN\",\n" +
                          "  \"salary\":\"5000\",\n" +
                          "  \"age\":\"35\"\n" +
                          "}";
          
                  given()
                          .header("Content-type", "application/json;charset=utf-8")
                          .body(requestBody)
                          .post()
                          .then()
                          .assertThat()
                          .body("status", equalTo("success"))
                          .body("data.name", equalTo("MITHUN"))
                          .body("data.salary", equalTo("5000"))
                          .body("data.age", equalTo("35"));
          
          
              }
          }
