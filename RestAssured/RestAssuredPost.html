Rest Assured POST Request


1. Go to WebsiteSampleAPI,Where we can get dummy service to test Web Service POST.




2. Click on details button corresponding to POST Method.




3. Always remember that post request need request body. This can be in the form of JSON/XML/gZIP. Copy request URL and JSON body from website.


4. Check below screenshot where Request URL: http://dummy.restapiexample.com/api/v1/create Request Method: POST Request Body: {"name":"test","salary":"123","age":"23"} 




    
     package RestAssured;
    
    import io.restassured.RestAssured.*;
    import io.restassured.matcher.RestAssuredMatchers.*;
    import org.hamcrest.Matchers.*;
    import io.restassured.RestAssured;
    import io.restassured.response.Response;
    
    public class RestTest {
    
    
        public static void main(String[] args) {
    
    
            RestAssured.baseURI="http://dummy.restapiexample.com/api/v1/create"; // To initiate the service...
    
            Response res = RestAssured
                    .given()
                    .body("{\"name\":\"name_001\",\"salary\":\"53001\",\"age\":\"23\"}" + "").post();
    
    
            System.out.println("Status Code is =="+ res.getStatusCode());
            System.out.println("Status Code is =="+ res.getBody().asString());
        }
    
    }
