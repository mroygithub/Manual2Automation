    package RestAssured;
    
    import io.restassured.RestAssured.*;
    import io.restassured.matcher.RestAssuredMatchers.*;
    import org.hamcrest.Matchers.*;
    import io.restassured.RestAssured;
    import io.restassured.response.Response;
    
    public class RestTest {
    
    
        public static void main(String[] args) {
    
    
            int employeeID = 0;
            RestAssured.baseURI = "http://dummy.restapiexample.com/api/v1/create";
    
            String requestBody = "{\n" +
                    "  \"name\":\"MITHUN\",\n" +
                    "  \"salary\":\"5000\",\n" +
                    "  \"age\":\"35\"\n" +
                    "}";
    
            Response response = null;
    
            try {
    
                response = RestAssured.given()
                        .header("Content-type", "application/json")
                        .header("host-header","c2hhcmVkLmJsdWVob3N0LmNvbQ==")
                        .body(requestBody)
                        .post();
    
            }
    
            catch (Exception e) {e.printStackTrace();}
    
            employeeID = Integer.valueOf(response.asString().split("id")[1].replaceAll("[^0-9]", ""));
    
            System.out.println(response.statusCode());System.out.println(response.asString());System.out.println(employeeID);
    
            // To delete the record
            RestAssured.baseURI = "http://dummy.restapiexample.com/api/v1/delete/"+String.valueOf(employeeID);
            System.out.println("http://dummy.restapiexample.com/api/v1/delete/"+String.valueOf(employeeID));
            response = RestAssured.given().when().delete();
            System.out.println(response.getStatusCode());
            System.out.println("Record Deleted Successfully");
        }
    }
