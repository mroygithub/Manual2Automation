TestNG Parameters Through XML

Pass application url using testng.xml[Test level parameter]

TestNG.xml will be written as below:

    <!DOCTYPE suite SYSTEM "https://testng.org/testng-1.0.dtd" >
    
    <suite name="Suite1" verbose="1" >
        <test name="manual2Automation" >
            <parameter name="url" value="https://www.google.com"></parameter>
            <classes>
                <class name="TestNGPKG.testNGClass" />
            </classes>
        </test>
    </suite>

TestNG class file will be written as below:

    
    package TestNGPKG;
    
    import org.testng.annotations.*;
    import org.openqa.selenium.*;
    import org.openqa.selenium.chrome.*;
    import java.util.concurrent.TimeUnit;
    
    public class testNGClass {
    
    
        WebDriver driver;
    
        @BeforeClass
        @Parameters({"url"})
        public void beforeClass(String url)
        {
            System. setProperty("webdriver.chrome.driver", "/Users/chromepath/Downloads/chromedriver2");
            driver = new ChromeDriver();
            driver.manage().window().maximize();
            driver.get(url);
            driver.manage().timeouts().pageLoadTimeout(40, TimeUnit.SECONDS);
        }
    
        @Test
        public void testMyApplication()
        {
            // To validate 'Google' Logo
            if(driver.findElement(By.id("hplogo")).isDisplayed())
            {System.out.println("Wikipedia Logo is available");}
        }
    
        @AfterClass
        public void afterClass()
        {
            driver.quit(); // To Close the Application....
        }
    }


Parameter name 'url' is case sensative and should be same both on xml and class file.

Pass multiple test level parameters using testng.xml

TestNG.xml will be written as below:

    span style="color:#e8bf6a;"><!DOCTYPE suite SYSTEM "https://testng.org/testng-1.0.dtd" >
    
    <suite name="Suite1" verbose="1" >
    
        <test name="manual2Automation" >
            <parameter name="url" value="https://www.google.com"></parameter>
            <parameter name="user" value="user1"></parameter>
            <parameter name="pass" value="pass1"></parameter>
            <classes>
                <class name="TestNGPKG.testNGClass" />
            </classes>
        </test>
    </suite>

TestNG class file will be written as below:

    package TestNGPKG;
    
    import org.testng.annotations.*;
    import org.openqa.selenium.*;
    import org.openqa.selenium.chrome.*;
    import sun.security.util.Password;
    
    import java.util.concurrent.TimeUnit;
    
    public class testNGClass {
    
    
        WebDriver driver;
    
        @BeforeClass
        @Parameters({"url"})
        public void beforeClass(String url)
        {
            System. setProperty("webdriver.chrome.driver", "/Users/chromepah/Downloads/chromedriver2");
            driver = new ChromeDriver();
            driver.manage().window().maximize();
            driver.get(url);
            driver.manage().timeouts().pageLoadTimeout(40, TimeUnit.SECONDS);
        }
    
        @Test
        @Parameters({"user","pass"})
        public void testMyApplication(String username , String password)
        {
            System.out.println("Application user id :"+ username);
            System.out.println("Application user id :"+ password);
            // To validate 'Google' Logo
            if(driver.findElement(By.id("hplogo")).isDisplayed())
            {System.out.println("Wikipedia Logo is available");}
        }
    
        @AfterClass
        public void afterClass()
        {
            driver.quit(); // To Close the Application....
        }
    }


Pass multiple suite level parameters using testng.xml

Suite level parameter is useful when same parameters are being used by multiple tests. TestNG.xml will be written as below:

        <!DOCTYPE suite SYSTEM "https://testng.org/testng-1.0.dtd" >
        
        <suite name="Suite1" verbose="1" >
        
            <parameter name="url" value="https://www.google.com"></parameter>
            <parameter name="user" value="user1"></parameter>
            <parameter name="pass" value="pass1"></parameter>
        
        
            <test name="manual2Automation" >
        
                <classes>
                    <class name="TestNGPKG.testNGClass" />
                </classes>
            </test>
        </suite>

TestNG class file will be written as below:

package TestNGPKG;

import org.testng.annotations.*;
import org.openqa.selenium.*;
import org.openqa.selenium.chrome.*;
import sun.security.util.Password;

import java.util.concurrent.TimeUnit;

public class testNGClass {


    WebDriver driver;



    @Test
    @Parameters({"url","user","pass"})
    public void testcase_1(String applicationurl , String username , String password)
    {
        System.out.println("Application user id :"+ applicationurl);
        System.out.println("Application user id :"+ username);
        System.out.println("Application user id :"+ password);

    }
    @Test
    @Parameters({"url","user","pass"})
    public void testcase_2(String applicationurl ,String username , String password)
    {
        System.out.println("Application user id :"+ applicationurl);
        System.out.println("Application user id :"+ username);
        System.out.println("Application user id :"+ password);

    }

}
