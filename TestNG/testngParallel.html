TestNG Parallel Execution

Run multiple methods using Parallel execution

TestNG.xml will be written as below:

    <!DOCTYPE suite SYSTEM "https://testng.org/testng-1.0.dtd" >
    
    <suite name="manual2Automation Test Suite" parallel="methods" thread-count="2">
    
        <test name="manual2Automation" >
    
            <classes>
                <class name="TestNGPKG.testNGClass" />
            </classes>
        </test>
    
    </suite>

Run multiple tests using Parallel execution

TestNG.xml will be written as below , where manual2Automation1 & manual2Automation2 tests will be executed at same time then manual2Automation3 & manual2Automation4. 

    <!DOCTYPE suite SYSTEM "https://testng.org/testng-1.0.dtd" >
    
    <suite name="manual2Automation Test Suite" parallel="tests" thread-count="2">
    
        <test name="manual2Automation1" >
            <classes>
                <class name="TestNGPKG.testNGClass">
                <methods><include name="testcase_1"/></methods></class>
            </classes>
        </test>
        <test name="manual2Automation2" >
            <classes>
                <class name="TestNGPKG.testNGClass">
                    <methods><include name="testcase_2"/></methods></class>
            </classes>
        </test>
        <test name="manual2Automation3" >
            <classes>
                <class name="TestNGPKG.testNGClass">
                    <methods><include name="testcase_3"/></methods></class>
            </classes>
        </test>
        <test name="manual2Automation4" >
            <classes>
                <class name="TestNGPKG.testNGClass">
                    <methods><include name="testcase_4"/></methods></class>
            </classes>
        </test>
    </suite>

Run multiple Class using Parallel execution

TestNG.xml will be written as below , where testNGClass1 & testNGClass2 tests will be executed at same time then testNGClass3 & testNGClass4. 

<!DOCTYPE suite SYSTEM "https://testng.org/testng-1.0.dtd" >
                
                <suite name="manual2Automation Test Suite" parallel="classes" thread-count="2">
                
                    <test name="manual2Automation1" >
                        <classes>
                            <class name="TestNGPKG.testNGClass1"/>
                            <class name="TestNGPKG.testNGClass2"/>
                            <class name="TestNGPKG.testNGClass3"/>
                            <class name="TestNGPKG.testNGClass4"/>
                        </classes>
                    </test>
                
                </suite>
