What is Assertion in TestNG?


All vadidations during testing are done by Assertions. This is checklist during test flow which says whether your test is pass or fail.




Example : Assert.assertTrue(condition)



Accepts boolean arguments and checks that a condition is true, If it isn't, throws an AssertionError.

            package TestNGPKG;
            
            import org.openqa.selenium.By;
            import org.openqa.selenium.WebDriver;
            import org.openqa.selenium.chrome.ChromeDriver;
            import org.testng.Assert;
            import org.testng.annotations.Test;
            
            public class testNGnew {
            
                WebDriver driver;
            
                @Test
                public void Example_AssertTrue()
            
                {
            
                    System. setProperty("webdriver.chrome.driver", "/Users/path/Downloads/chromedriver2");
                    driver = new ChromeDriver();
                    driver.get("https://www.google.com");
                    Assert.assertTrue(driver.findElement(By.id("hplogo")).isDisplayed());
                    driver.quit();
                    System.out.println("Google Application Closed Successfully");
                }
            }

Example : Assert.assertFalse(condition)



Accepts boolean arguments and checks that a condition is false, If it isn't, throws an AssertionError.

    
    package TestNGPKG;
    
    import org.openqa.selenium.By;
    import org.openqa.selenium.WebDriver;
    import org.openqa.selenium.chrome.ChromeDriver;
    import org.testng.Assert;
    import org.testng.annotations.AfterClass;
    import org.testng.annotations.Test;
    
    public class testNGnew {
    
        static WebDriver driver;
    
        @Test
        public void Example_AssertTrue()
    
        {
    
            System. setProperty("webdriver.chrome.driver", "/Users/path/Downloads/chromedriver2");
            driver = new ChromeDriver();
            driver.get("https://www.google.com");
            Assert.assertFalse(googleWrongLogo());
            System.out.println("Google Application Closed Successfully");
        }
    
        public static boolean googleWrongLogo()
        {
    
            boolean condition = false;
            try{
                if(driver.findElement(By.id("hplogo1")).isDisplayed())
                {
                    condition = true;
                }
            }
            catch(Exception e){e.printStackTrace();}
    
            return condition;
        }
    
        @AfterClass
        public void teardown()
        {
            driver.quit();
        }
    
    }
