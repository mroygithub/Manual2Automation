TestNG Priority and Groups

We can set priority in test method itself as below:

TestNG.xml will be written as below:

    package TestNGPKG;
    
    import org.openqa.selenium.WebDriver;
    import org.openqa.selenium.chrome.ChromeDriver;
    import org.testng.annotations.Test;
    
    import java.util.concurrent.TimeUnit;
    
    public class testNGClass2 {
    
        @Test
        public void testcase_3()
        {
            System.out.println("Third TC");
        }
    
        @Test
        public void testcase_2()
        {
            System.out.println("Second TC");
        }
    
    
        @Test
        public void testcase_1()
        {
            System.out.println("First TC");
        }
    
    
    }


Output will be displayed as below:
[TestNG] Running:
/Users/path/Library/Caches/IdeaIC2018.3/temp-testng-customsuite.xml
First TC
Second TC
Third TC



Set priority using 'priority' key as below:
    package TestNGPKG;
    
    import org.openqa.selenium.WebDriver;
    import org.openqa.selenium.chrome.ChromeDriver;
    import org.testng.annotations.Test;
    
    import java.util.concurrent.TimeUnit;
    
    public class testNGClass2 {
    
        @Test(priority = 1)
        public void testcase_3()
        {
            System.out.println("Third TC");
        }
    
        @Test(priority = 2)
        public void testcase_2()
        {
            System.out.println("Second TC");
        }
    
    
        @Test(priority = 3)
        public void testcase_1()
        {
            System.out.println("First TC");
        }
        @Test(priority = 0)
        public void testcase_0()
        {
            System.out.println("ZERO TC");
        }
    
    
    }

Result :
[TestNG] Running:
/Users/temp-testng-customsuite.xml
ZERO TC
Third TC
Second TC
First TC Note : Priority key overlaps method level priority.

Set priority using 'priority' key with -ve value:
    package TestNGPKG;
    
    import org.openqa.selenium.WebDriver;
    import org.openqa.selenium.chrome.ChromeDriver;
    import org.testng.annotations.Test;
    
    import java.util.concurrent.TimeUnit;
    
    public class testNGClass2 {
    
        @Test(priority = 2)
        public void sign()
        {
            System.out.println("2 TC");
        }
    
        @Test(priority = 1)
        public void test()
        {
            System.out.println("1 TC");
        }
    
    
        @Test(priority = 0)
        public void cases()
        {
            System.out.println("0 TC");
        }
        @Test(priority = -1)
        public void hello()
        {
            System.out.println("-1 TC");
        }
    
    
    }

Result :

-1 TC
0 TC
1 TC

TestNG Grouping of Test Cases:

TestNG class as below :

    
    package TestNGPKG;
    
    import org.openqa.selenium.WebDriver;
    import org.openqa.selenium.chrome.ChromeDriver;
    import org.testng.annotations.Test;
    
    import java.util.concurrent.TimeUnit;
    
    public class testNGClass2 {
    
        @Test(groups= {"Smoke"})
        public void sign()
        {
            System.out.println("sign TC..this is Smoke");
        }
    
        @Test(groups= {"Regression"})
        public void test()
        {
            System.out.println("test TC..this is Regression");
        }
    
    
        @Test(groups= {"Smoke"})
        public void cases()
        {
            System.out.println("cases TC..this is Smoke");
        }
        @Test(groups= {"Regression"})
        public void hello()
        {
            System.out.println("hello TC..this is Regression");
        }
    
    
    }

To run only SMOKE tests TestNG XML as below :

    
    <!DOCTYPE suite 
    SYSTEM "https://testng.org/testng-1.0.dtd" >
    
    <suite name="manual2Automation Test Suite" parallel="false" thread-count="1">
    
        <test name="manual2Automation1" >
            <groups>
                <run>
                    <include name="Smoke"/>
                </run>
            </groups>
            <classes>
                <class name="TestNGPKG.testNGClass2"/>
            </classes>
        </test>
    
    </suite>


Test Result as :
[TestNG] Running:
/Users/main/resources/myTestNG.xml
cases TC..this is Smoke
sign TC..this is Smoke

We can Exclude a perticular group also:[below example is excluding 'Smoke' , which will execute only Regression group of tests.]

    
    <!DOCTYPE suite SYSTEM "https://testng.org/testng-1.0.dtd" >
    
    <suite name="manual2Automation Test Suite" parallel="classes" thread-count="2">
    
        <test name="manual2Automation1" >
            <groups>
                <run>
                    <exclude name="Smoke"/>
                </run>
            </groups>
            <classes>
                <class name="TestNGPKG.testNGClass2"/>
            </classes>
        </test>
    
    </suite>
