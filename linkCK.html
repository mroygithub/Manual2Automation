        public void checksOnlyLinks()
        {

        public static void verifyLinkActive(String linkUrl , int a  , String type)
        {
    
            boolean FLAG = false;
            z=a;
            try
            {
    
                URL url = new URL(linkUrl);
                HttpURLConnection httpURLConnect=(HttpURLConnection)url.openConnection();
                httpURLConnect.setConnectTimeout(3000);
                httpURLConnect.connect();
    
                if(httpURLConnect.getResponseCode()==301 || httpURLConnect.getResponseCode()==302)
                {
                    ResponseCode = String.valueOf(httpURLConnect.getResponseCode());
                    WarningResult=linkUrl+" - "+httpURLConnect.getResponseMessage()+"<=Permanent URL redirection=>";
                    Type=type;
                    resultoutput_warning();
                }
                else if(httpURLConnect.getResponseCode()==200)
                {
    
                    ResponseCode = String.valueOf(httpURLConnect.getResponseCode());
                    PassResult=linkUrl+" - "+httpURLConnect.getResponseMessage();
                    Type=type;
                    resultoutput_pass();
    
                }
                else if(httpURLConnect.getResponseCode()== HttpURLConnection.HTTP_NOT_FOUND)
                {
                    FailResult=linkUrl+" - "+httpURLConnect.getResponseMessage() + " - "+ HttpURLConnection.HTTP_NOT_FOUND;
                    ResponseCode = String.valueOf(httpURLConnect.getResponseCode());
                    Type=type;
                    resultoutput_fail();
                }
                else
                {
                    FailResult=linkUrl+"<===ISSUES===> Please check the link";
                    ResponseCode = String.valueOf(httpURLConnect.getResponseCode());
                    Type=type;
                    resultoutput_fail();
                }
    
            } catch (Exception e) {
                System.out.println("The Link Has Issue ==>"+linkUrl );
            }
        }
